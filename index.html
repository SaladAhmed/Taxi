Dakhliga (Income) if(d.type === "Income-Sentral") { sentral += d.amount; // Taxi MVA waa 12% (12/112) mva_refund += (d.amount * 12 / 112); } else if(d.type === "Income-Own") { own += d.amount; } // 2. Kharashka (Expenses) else { kharash_total += d.amount; // Kharashyada leh MVA 25% (Regnskap, Biler, Bom) // Xisaabta: (Amount * 25 / 125) ama si fudud (Amount / 5) if(d.type === "Kharash-Biler" || d.type === "Kharash-Bom" || d.type === "Kharash-Regn") { mva_refund += (d.amount / 5); } } // Naqshadda Card-ka let borderColor = d.type.includes('Income') ? '#2ecc71' : '#ef4444'; if(d.type === "Income-Own") borderColor = '#3b82f6'; list.innerHTML += `
${d.type.split('-')[1].toUpperCase()}

${d.category}

${d.date}
${d.amount.toLocaleString()} kr
Delete
`; }); // Isku darka Dashboard-ka const net = (sentral + own) - kharash_total; document.getElementById('total-sentral').innerText = sentral.toLocaleString() + " kr"; document.getElementById('total-own').innerText = own.toLocaleString() + " kr"; document.getElementById('total-kharash').innerText = kharash_total.toLocaleString() + " kr"; document.getElementById('mva-val').innerText = Math.floor(mva_refund).toLocaleString() + " kr"; document.getElementById('ferie-val').innerText = Math.floor((sentral + own) * 0.102).toLocaleString() + " kr"; document.getElementById('net-result').innerText = net.toLocaleString() + " kr"; }); }
