// Qaybtan koodhka ah ka beddel function-ka loadData() ee ku dhex jira koodhkaaga function loadData() { onSnapshot(query(collection(db, "xisaabaadka"), orderBy("date", "desc")), (snapshot) => { const list = document.getElementById('transactionList'); const statsList = document.getElementById('categoryTotals'); list.innerHTML = ''; statsList.innerHTML = ''; let ti=0, te=0, ownerSum=0, mvaSum=0, totalFP=0; let catTotals = {}; // Dictionary-ga turjumaada xogta database-ka const translateCat = (catName) => { const map = { // SO -> EN/NO 'Shidaal': { en: 'Fuel', no: 'Drivstoff' }, 'Drivstoff': { so: 'Shidaal', en: 'Fuel' }, 'Dayactir': { en: 'Repair', no: 'Reparasjon' }, 'Mushaar (Lønn)': { en: 'Salary', so: 'Mushaar' }, 'Bankuttak (Owner)': { en: 'Withdrawal', no: 'Eieruttak' }, 'Regnskap': { so: 'Xisaabiye', en: 'Accounting' }, 'Forsikring': { so: 'Caymis', en: 'Insurance' }, 'Wax kale': { en: 'Other', no: 'Annet' } }; if (currentLang === 'so') return catName; // Haddii ay Soomaali tahay sideeda u daa return (map[catName] && map[catName][currentLang]) ? map[catName][currentLang] : catName; }; snapshot.forEach((docSnap) => { const d = docSnap.data(); const translatedType = translateCat(d.type); // Halkan ayuu ku turjumayaa if(d.mode === 'income') ti += d.amount; else { te += d.amount; if(d.type.includes('Owner') || d.type.includes('Eier')) ownerSum += d.amount; if(d.type.includes('Lønn') || d.type.includes('Salary')) totalFP += (d.amount * 0.102); if(['Drivstoff','Fuel','Shidaal','Dayactir','Repair','Regnskap','Accounting'].some(word => d.type.includes(word))) mvaSum += (d.amount / 5); } catTotals[translatedType] = (catTotals[translatedType] || 0) + d.amount; // Qaybta Guudmar (Dashboard) list.innerHTML += `

${d.name}

${translatedType}

${d.date}

${Math.round(d.amount).toLocaleString()} kr

TIRTIR
`; }); // Qaybta Warbixinta (Rapport) for (const [cat, total] of Object.entries(catTotals)) { const isInc = snapshot.docs.find(doc => translateCat(doc.data().type) === cat)?.data().mode === 'income'; statsList.innerHTML += `
${cat}${Math.round(total).toLocaleString()} kr
`; } // Cusboonaysii dusha sare (Totals) document.getElementById('totalIncome').innerText = Math.round(ti).toLocaleString() + ' kr'; document.getElementById('totalExpense').innerText = Math.round(te).toLocaleString() + ' kr'; document.getElementById('totalOwner').innerText = Math.round(ownerSum).toLocaleString() + ' kr'; document.getElementById('mvaTax').innerText = Math.round(mvaSum).toLocaleString() + ' kr'; document.getElementById('ferieTax').innerText = Math.round(totalFP).toLocaleString() + ' kr'; const net = ti - te - totalFP; document.getElementById('netResult').innerText = Math.round(net).toLocaleString() + ' kr'; }); }
