function loadData() { db.ref('users/' + auth.currentUser.uid + '/entries').on('value', snap => { let ti = 0, te = 0; let exp7000 = 0; let exp7020 = 0; let exp7040 = 0; let exp6900 = 0; let exp6755 = 0; let expOther = 0; const list = document.getElementById('entries-list'); const reportBody = document.getElementById('report-body'); list.innerHTML = ""; reportBody.innerHTML = ""; snap.forEach(child => { const v = child.val(); const key = child.key; if (v.mainCat === 'income') { ti += v.amt; } else { te += v.amt; if(v.subCat === '7000'){ exp7000 += v.amt; } else if(v.subCat === '7020'){ exp7020 += v.amt; } else if(v.subCat === '7040'){ exp7040 += v.amt; } else if(v.subCat === '6900'){ exp6900 += v.amt; } else if(v.subCat === '6755'){ exp6755 += v.amt; } else{ expOther += v.amt; } } list.innerHTML += `
${v.subCat} - ${v.desc}
${v.date}
${v.amt.toLocaleString('no-NO',{minimumFractionDigits:2})} kr
`; reportBody.innerHTML += ` ${v.date} ${v.subCat} ${v.desc} ${v.amt.toLocaleString('no-NO',{minimumFractionDigits:2})} kr `; }); document.getElementById('total-income').innerText = ti.toLocaleString('no-NO') + " kr"; document.getElementById('total-expense').innerText = te.toLocaleString('no-NO') + " kr"; document.getElementById('total-netto').innerText = (ti - te).toLocaleString('no-NO') + " kr"; document.getElementById('res-total').innerText = (ti - te).toLocaleString('no-NO') + " kr"; document.getElementById('res-total').style.color = (ti - te) >= 0 ? 'var(--green)' : 'var(--red)'; const mva = Math.round(ti * 0.12); document.getElementById('res-mva').innerText = mva.toLocaleString('no-NO') + " kr"; document.getElementById('res-avsk').innerText = "0 kr"; // ===== UTGIFTER TOTAL ===== document.getElementById('expense-total-big').innerText = te.toLocaleString('no-NO') + " kr"; document.getElementById('exp-7000').innerText = exp7000.toLocaleString('no-NO') + " kr"; document.getElementById('exp-7020').innerText = exp7020.toLocaleString('no-NO') + " kr"; document.getElementById('exp-7040').innerText = exp7040.toLocaleString('no-NO') + " kr"; document.getElementById('exp-6900').innerText = exp6900.toLocaleString('no-NO') + " kr"; document.getElementById('exp-6755').innerText = exp6755.toLocaleString('no-NO') + " kr"; document.getElementById('exp-other').innerText = expOther.toLocaleString('no-NO') + " kr"; }); }
